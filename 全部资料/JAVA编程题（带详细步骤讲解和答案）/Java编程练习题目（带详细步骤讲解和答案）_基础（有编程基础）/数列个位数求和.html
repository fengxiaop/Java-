public class MainClass {

    public static void main(String args[]) {

        String numberStr = "105;20;30;40;50;55;26;7";

        System.out.println("原字符数列："+numberStr);

        System.out.println("数列个位数求和结果："+sum(numberStr));

    }

    /**

     * 数列个位数求和

     * @param numberStr 求和数列

     * @return 求和结果

     */

    public static int sum(String numberStr) {

        //分割数列

        String[] numbers = numberStr.split(";");

        int ret = 0;

        for (int i = 0; i < numbers.length; i++) {

            String num = numbers[i];

            // 分割个位数

            num = num.substring(num.length()-1);

            // 转型并合计

            ret += Integer.valueOf(num);

        }

        return ret;

    }

}
