public class MainClass {

    public static void main(String[] args) {

        //随机产生200个用户的年龄,年龄范围：0 ~ 60岁

        int[] ages = getAges();

        //计算成年人的数量（年龄大于等于18岁）

        int adults = getAdultCount(ages);

        System.out.println("200个用户中，共有" + adults + "个成年人");

    }


    /**

     * 随机产生200个用户的年龄,年龄范围：0 ~ 60岁

     * @return 200个用户的年龄

     */

    public static int[] getAges() {

        //200个用户的年龄数组

        int[] ages = new int[200];

        for (int i = 0; i < ages.length; i++) {

            //随机产生200个用户的年龄,年龄范围：0 ~ 60岁

            ages[i] = (int) (Math.random() * 61);

        }

        return ages;

    }

    /**

     * 计算成年人的数量（年龄大于等于18岁）

     * @param ages 随机产生的200个用户的年龄

     * @return 成年人的数量

     */

    public static int getAdultCount(int[] ages) {

        int adult = 0;

        for (int i = 0; i < ages.length; i++) {

            if (ages[i] >= 18)

                adult++;

        }

        return adult;

    }

}
