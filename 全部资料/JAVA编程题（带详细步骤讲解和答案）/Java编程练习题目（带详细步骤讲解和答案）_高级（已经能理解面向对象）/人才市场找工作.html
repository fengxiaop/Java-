1、实验要求

    1-1. 业务说明：

        1-1.1. 本实验以大学毕业生找工作为业务背景。

        1-1.2. 大学生有不同的专业背景，但通常都需要找工作。 

        1-1.3. 人才市场根据不同的专业和雇主要求为求职者提供工作机会。

        1-1.4. 人才市场需要每类专业的毕业生提供专业平均分和姓名。

            2）限制2. 方法设定为0参的公共方法；

public interface SeekJob {

    public static float SEEKER_AVERAGE_SCORE = 85.0f;

    /**

     * 由接口实现类提供

     * @return 求职者姓名

     */

    public String getName() ;

    /**

     * 不同的专业的专业课程构成项不同，具体构成由实现类提供。

     * 计算专业课的平均分的算法可不相同，如加权平均、比重累加等。

     * @return 获取专业课总平均分

     */

    public float getSeekerAverage();

}

public class HrMarketer {


    public HrMarketer() {

    }

    //人才市场服务的对象：找工作的毕业生

    public String seekJob(SeekJob seeker) {

        String result = "";

        float average = seeker.getSeekerAverage();

        if (average > SeekJob.SEEKER_AVERAGE_SCORE) {

            result = seeker.getName() + "被本公司录用。";

        } else {

            result = seeker.getName() + "成绩未达标，不予录用。";

        }

        return result;

    }

}
