public interface ShapeCalc {

    /**

     * 计算图形的周长和面积

     */

    public float calcPerim();

    public float calcArea();

   

}

 代码清单：cn.campsg.java.experiment.impl.SquareCalImpl.java

public class SquareCalImpl implements ShapeCalc {

   

    private float length;

   

    public float getLength() {

        return length;

    }

 

    public void setLength(float length) {

        this.length = length;

    }

 

    public SquareCalImpl(){}

   

    @Override

    public float calcPerim() {

        float peri = this.getLength()*4;

        return peri;

    }

 

    @Override

    public float calcArea() {

        float area = this.getLength() * this.getLength() ;

        return area;

    }

 

}
