2. 代码清单：cn.campsg.java.experiment.entity.Rose.java

public class Rose {

    //预定要送的玫瑰花总数，默认

    private int total=9 ;

    public Rose() {

    }


    public Rose(int total) {

        this.total = total ;

    }


    /**

     * 实际要送出玫瑰花

     * @param num 送出的玫瑰花数

     * @throws RoseException 玫瑰花数量多于预送数量时抛出的异常

     */

    public void giveRose(int num) throws RoseException {

        // 若送出的数量小于1，则退出方法的执行

        if(num<=0){

            System.out.println("送花的数量不正确：数量不能小于零！");

            return ;

        }

        if (total<=0||num > this.total) {

            throw new RoseException("没钱买"+num+"只玫瑰花，仅能买"+total+"只玫瑰花了。对不起！");

        }else {

            System.out.println("亲爱的，送你" + num + "朵玫瑰花！");

            this.total-=num;

        }

    }

}
