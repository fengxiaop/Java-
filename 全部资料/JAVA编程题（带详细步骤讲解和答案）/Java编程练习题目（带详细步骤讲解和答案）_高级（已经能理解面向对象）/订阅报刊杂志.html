  16版.订阅报刊杂志

1、实验要求

    本实验要求：以读者订阅报刊杂志为背景，体验多线程生产/订阅模式下的经典应用。

    1-1. 业务说明：

        1-1.1. 本实验以实际生活中报刊杂志的订阅为业务背景。

        1-1.2. 订阅者通常有很多种，他们可以同时向出版社发出订阅申请。 

        1-1.3. 出版社通常根据自身的资源条件对收到的订阅申请做统一的处理。

        1-1.4. 出版社通常能接受并处理的订阅申请量是有一定限度的。

        1-1.5. 出版社汇总订阅申请信息后交由独立业务单元处理，直至完毕。

        1-1.6. 本业务主要是模拟多读者同时订阅、信息汇总后再行处理的业务情景。

        1-1.7. 本实验的关键业务概念：

                1） 订阅者，即订阅信息生产者，向共享池中放入订阅信息；

                2） 信息汇总，即订阅信息共享池，生产者和消费者共用；

                3） 业务单元，即订阅信息消费者，处理订阅者生产的订阅信息。

public class SharePool {

    //订阅共享池

    private List<String> pool;

    //最大订阅处理量

    private final int  MAX = 15;

 

    public List<String> getPool() {

       return pool;

    }

    public void setPool(List<String> pool) {

       this.pool = pool;

    }

    public SharePool() {

       this.pool = new ArrayList<String>(MAX);

    }

    /**

     * 申请者处理:默认订阅池只能放置15个订阅请求,当池满时则不能提交订阅请求。

     *

     * @param media

     *            申请订阅内容

     */

    public void produce(String media) {

       synchronized (this) {

           if (pool.size() == MAX) {//控制不超过最大处理量

              try {

                  System.out.println("订阅请求队列已满，等待系统处理订阅请求中……");

                  wait();

              } catch (InterruptedException e) {

                  e.printStackTrace();

              }

           }

           pool.add(media); // 每次插入一个元素

           System.out.println("订阅者@"+Thread.currentThread().getName()+":订阅《" + media + "》申请已提交.当前订阅数量为：" + pool.size());

           notify();

       }

    }

    /**

     * 处理订阅请求：生产者处理

     */

    public void consume() {

       synchronized (this) {

           if (pool.size() == 0) {

              try {

                  System.out.println("处理者@"+Thread.currentThread().getName()+"暂无订阅请求信息，等待中……");

                  wait();                 

              } catch (InterruptedException e) {

                  e.printStackTrace();

              }

           }

           if (!pool.isEmpty()) {

              String media = pool.remove(0); // 每次移走队首元素

              System.out.println("处理者@"+Thread.currentThread().getName()+":处理《" + media + "》 订阅已完毕。尚待处理订阅数量为：" + pool.size());

           }

           notify();

       }

    }

 

}

              System.out.println("订阅者@"+Thread.currentThread().getName()+" 在提交第 "+counter+" 份订阅申请。");

              pool.produce(media+counter); // 每次插入一个元素

              count--;

              try {

              }     

           }

       int counter = 0 ;

       while (true) {

           counter++ ;

           //独立一个线程只负责处理15个请求，处理完了退出。：可以换成多线程处理，模仿生产者模式进行。练习者自行处理

           if(counter>15){

              System.out.println("本线程完成订阅处理量，即刻退出。处理者@"+Thread.currentThread().getName());

              break ;

           }

           System.out.println("处理者@"+Thread.currentThread().getName()+":处理第 "+counter+"份订阅。");

           this.pool.consume();

       }

    }
